129  int numPartitions,
130  int replicationFactor,
131  const Properties& topicConfig,

141  if (!rkNewTopics.back())
142  {
143  return admin::CreateTopicsResult(Error{RD_KAFKA_RESP_ERR__INVALID_ARG, rd_kafka_err2str(RD_KAFKA_RESP_ERR__INVALID_ARG)});
144  }
145 
146  for (const auto& conf: topicConfig.map())
147  {
148  rd_kafka_resp_err_t err = rd_kafka_NewTopic_set_config(rkNewTopics.back().get(), conf.first.c_str(), conf.second.c_str());
149  if (err != RD_KAFKA_RESP_ERR_NO_ERROR)
150  {
151  std::string errMsg = "Invalid config[" + conf.first + "=" + conf.second + "]";
152  KAFKA_API_DO_LOG(Log::Level::Err, errMsg.c_str());
153  return admin::CreateTopicsResult(Error{RD_KAFKA_RESP_ERR__INVALID_ARG, errMsg});
154  }
155  }
156  }
157 
158  std::vector<rd_kafka_NewTopic_t*> rk_topics;
159  rk_topics.reserve(rkNewTopics.size());
160  for (const auto& topic : rkNewTopics) { rk_topics.emplace_back(topic.get()); }
161 
162  auto rk_queue = rd_kafka_queue_unique_ptr(rd_kafka_queue_new(getClientHandle()));
163 
164  rd_kafka_CreateTopics(getClientHandle(), rk_topics.data(), rk_topics.size(), nullptr, rk_queue.get());
165 
166  auto rk_ev = rd_kafka_event_unique_ptr();
167 
168  const auto end = std::chrono::steady_clock::now() + timeout;
169  do
170  {
171  rk_ev.reset(rd_kafka_queue_poll(rk_queue.get(), EVENT_POLLING_INTERVAL_MS));
172 
173  if (rd_kafka_event_type(rk_ev.get()) == RD_KAFKA_EVENT_CREATETOPICS_RESULT) break;
174 
175  if (rk_ev)
176  {
177  KAFKA_API_DO_LOG(Log::Level::Err, "rd_kafka_queue_poll got event[%s], with error[%s]", rd_kafka_event_name(rk_ev.get()), rd_kafka_event_error_string(rk_ev.get()));
178  rk_ev.reset();
179  }
180  } while (std::chrono::steady_clock::now() < end);
181 
182  if (!rk_ev)
183  {
184  return admin::CreateTopicsResult(Error{RD_KAFKA_RESP_ERR__TIMED_OUT, "No response within the time limit"});
185  }
186 
187  std::list<Error> errors;
188 
189  if (rd_kafka_resp_err_t respErr = rd_kafka_event_error(rk_ev.get()))
190  {
191  errors.emplace_back(respErr, rd_kafka_event_error_string(rk_ev.get()));
192  }
193 
194  // Fetch per-topic results
195  const rd_kafka_CreateTopics_result_t* res = rd_kafka_event_CreateTopics_result(rk_ev.get());
196  std::size_t res_topic_cnt{};
197  const rd_kafka_topic_result_t** res_topics = rd_kafka_CreateTopics_result_topics(res, &res_topic_cnt);
198 
199  errors.splice(errors.end(), getPerTopicResults(res_topics, res_topic_cnt));
200 
201  // Return the error if any
202  if (!errors.empty())
203  {
204  return admin::CreateTopicsResult{combineErrors(errors)};
205  }
206 
207  // Update metedata
208  do
209  {
210  auto listResult = listTopics();
211  if (!listResult.error)
212  {
213  return admin::CreateTopicsResult(Error{RD_KAFKA_RESP_ERR_NO_ERROR, "Success"});
214  }
215  } while (std::chrono::steady_clock::now() < end);
216 
217  return admin::CreateTopicsResult(Error{RD_KAFKA_RESP_ERR__TIMED_OUT, "Updating metadata timed out"});
218 }
