397  }

427 {
428  _opened = false;
429 
430  stopBackgroundPollingIfNecessary();

432  try
433  {

460 inline void
461 KafkaConsumer::subscribe(const Topics& topics, consumer::RebalanceCallback rebalanceCallback, std::chrono::milliseconds timeout)
462 {
463  if (!_userAssignment.empty())
464  {
465  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__FAIL, "Unexpected Operation! Once assign() was used, subscribe() should not be called any more!"));
466  }
467 
468  if (isCooperativeEnabled() && topics == _userSubscription)
469  {
470  KAFKA_API_DO_LOG(Log::Level::Info, "skip subscribe (no change since last time)");
471  return;
472  }
473 
474  _userSubscription = topics;
475 
476  std::string topicsStr = toString(topics);
477  KAFKA_API_DO_LOG(Log::Level::Info, "will subscribe, topics[%s]", topicsStr.c_str());
478 
479  _rebalanceCb = std::move(rebalanceCallback);
480 
481  auto rk_topics = rd_kafka_topic_partition_list_unique_ptr(createRkTopicPartitionList(topics));
482 
483  Error result{ rd_kafka_subscribe(getClientHandle(), rk_topics.get()) };
484  KAFKA_THROW_IF_WITH_ERROR(result);
485 
486  _pendingEvent = PendingEvent::PartitionsAssignment;
487 
488  // The rebalcance callback would be served during the time (within this thread)
489  for (const auto end = std::chrono::steady_clock::now() + timeout; std::chrono::steady_clock::now() < end; )
490  {
491  rd_kafka_poll(getClientHandle(), EVENT_POLLING_INTERVAL_MS);
492 
493  if (!_pendingEvent)
494  {
495  KAFKA_API_DO_LOG(Log::Level::Notice, "subscribed, topics[%s]", topicsStr.c_str());
496  return;
497  }
498  }
499 
500  _pendingEvent.reset();
501  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__TIMED_OUT, "subscribe() timed out!"));
502 }

504 inline void
505 KafkaConsumer::unsubscribe(std::chrono::milliseconds timeout)
506 {
507  if (_userSubscription.empty() && _userAssignment.empty())
508  {
509  KAFKA_API_DO_LOG(Log::Level::Info, "skip unsubscribe (no assignment/subscription yet)");
510  return;
511  }
512 
513  KAFKA_API_DO_LOG(Log::Level::Info, "will unsubscribe");
514 
515  // While it's for the previous `assign(...)`
516  if (!_userAssignment.empty())
517  {
518  changeAssignment(isCooperativeEnabled() ? PartitionsRebalanceEvent::IncrementalUnassign : PartitionsRebalanceEvent::Revoke,
519  _userAssignment);
520  _userAssignment.clear();
521 
522  KAFKA_API_DO_LOG(Log::Level::Notice, "unsubscribed (the previously assigned partitions)");
523  return;
524  }
525 
526  _userSubscription.clear();
527 
528  Error result{ rd_kafka_unsubscribe(getClientHandle()) };
529  KAFKA_THROW_IF_WITH_ERROR(result);
530 
531  _pendingEvent = PendingEvent::PartitionsRevocation;
532 
533  // The rebalance callback would be served during the time (within this thread)
534  for (const auto end = std::chrono::steady_clock::now() + timeout; std::chrono::steady_clock::now() < end; )
535  {
536  rd_kafka_poll(getClientHandle(), EVENT_POLLING_INTERVAL_MS);
537 
538  if (!_pendingEvent)
539  {
540  KAFKA_API_DO_LOG(Log::Level::Notice, "unsubscribed");
541  return;
542  }
543  }
544 
545  _pendingEvent.reset();
546  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__TIMED_OUT, "unsubscribe() timed out!"));
547 }

561 inline void
562 KafkaConsumer::changeAssignment(PartitionsRebalanceEvent event, const TopicPartitions& tps)
563 {
564  auto rk_tps = rd_kafka_topic_partition_list_unique_ptr(createRkTopicPartitionList(tps));
565 
566  Error result;
567  switch (event)
568  {
569  case PartitionsRebalanceEvent::Assign:
570  result = Error{ rd_kafka_assign(getClientHandle(), rk_tps.get()) };
571  // Update assignment
572  _assignment = tps;
573  break;
574 
575  case PartitionsRebalanceEvent::Revoke:
576  result = Error{ rd_kafka_assign(getClientHandle(), nullptr) };
577  // Update assignment
578  _assignment.clear();
579  break;
580 
581  case PartitionsRebalanceEvent::IncrementalAssign:
582  result = Error{ rd_kafka_incremental_assign(getClientHandle(), rk_tps.get()) };
583  // Update assignment
584  for (const auto& tp: tps)
585  {
586  auto found = _assignment.find(tp);
587  if (found != _assignment.end())
588  {
589  std::string tpStr = toString(tp);
590  KAFKA_API_DO_LOG(Log::Level::Err, "incremental assign partition[%s] has already been assigned", tpStr.c_str());
591  continue;
592  }
593  _assignment.emplace(tp);
594  }
595  break;
596 
597  case PartitionsRebalanceEvent::IncrementalUnassign:
598  result = Error{ rd_kafka_incremental_unassign(getClientHandle(), rk_tps.get()) };
599  // Update assignment
600  for (const auto& tp: tps)
601  {
602  auto found = _assignment.find(tp);
603  if (found == _assignment.end())
604  {
605  std::string tpStr = toString(tp);
606  KAFKA_API_DO_LOG(Log::Level::Err, "incremental unassign partition[%s] could not be found", tpStr.c_str());
607  continue;
608  }
609  _assignment.erase(found);
610  }
611  break;
612  }
613 
614  KAFKA_THROW_IF_WITH_ERROR(result);
615 }

647 // Seek & Position
648 inline void
649 KafkaConsumer::seek(const TopicPartition& topicPartition, Offset offset, std::chrono::milliseconds timeout)
650 {
651  std::string topicPartitionStr = toString(topicPartition);
652  KAFKA_API_DO_LOG(Log::Level::Info, "will seek with topic-partition[%s], offset[%d]", topicPartitionStr.c_str(), offset);
653 
654  auto rkt = rd_kafka_topic_unique_ptr(rd_kafka_topic_new(getClientHandle(), topicPartition.first.c_str(), nullptr));
655  if (!rkt)
656  {
657  KAFKA_THROW_ERROR(Error(rd_kafka_last_error()));
658  }
659 
660  const auto end = std::chrono::steady_clock::now() + timeout;
661 
662  rd_kafka_resp_err_t respErr = RD_KAFKA_RESP_ERR_NO_ERROR;
663  do
664  {
665  respErr = rd_kafka_seek(rkt.get(), topicPartition.second, offset, SEEK_RETRY_INTERVAL_MS);
666  if (respErr != RD_KAFKA_RESP_ERR__STATE && respErr != RD_KAFKA_RESP_ERR__TIMED_OUT && respErr != RD_KAFKA_RESP_ERR__OUTDATED)
667  {
668  break;
669  }
670 
671  // If the "seek" was called just after "assign", there's a chance that the toppar's "fetch_state" (async setted) was not ready yes.
672  // If that's the case, we would retry again (normally, just after a very short while, the "seek" would succeed)
673  std::this_thread::yield();
674  } while (std::chrono::steady_clock::now() < end);
675 
676  KAFKA_THROW_IF_WITH_ERROR(Error(respErr));
677 
678  KAFKA_API_DO_LOG(Log::Level::Info, "seeked with topic-partition[%s], offset[%d]", topicPartitionStr.c_str(), offset);
679 }

701 inline std::map<TopicPartition, Offset>
702 KafkaConsumer::offsetsForTime(const TopicPartitions& topicPartitions,
703  std::chrono::time_point<std::chrono::system_clock> timepoint,
704  std::chrono::milliseconds timeout) const
705 {
706  if (topicPartitions.empty()) return TopicPartitionOffsets();
707 
708  auto msSinceEpoch = std::chrono::duration_cast<std::chrono::milliseconds>(timepoint.time_since_epoch()).count();
709 
710  auto rk_tpos = rd_kafka_topic_partition_list_unique_ptr(createRkTopicPartitionList(topicPartitions));
711 
712  for (int i = 0; i < rk_tpos->cnt; ++i)
713  {
714  rd_kafka_topic_partition_t& rk_tp = rk_tpos->elems[i];
715  // Here the `msSinceEpoch` would be overridden by the offset result (after called by `rd_kafka_offsets_for_times`)
716  rk_tp.offset = msSinceEpoch;
717  }
718 
719  Error error{ rd_kafka_offsets_for_times(getClientHandle(), rk_tpos.get(), static_cast<int>(timeout.count())) }; // NOLINT
720  KAFKA_THROW_IF_WITH_ERROR(error);
721 
722  auto results = getTopicPartitionOffsets(rk_tpos.get());
723 
724  // Remove invalid results (which are not updated with an valid offset)
725  for (auto it = results.begin(); it != results.end(); )
726  {
727  it = ((it->second == msSinceEpoch) ? results.erase(it) : std::next(it));
728  }
729 
730  return results;
731 }

738  for (const auto& topicPartition: topicPartitions)

809 {

822  // Store the offsets for all these polled messages (for "enable.auto.commit=true" case)
823  storeOffsetsIfNecessary(output);

843 inline void
844 KafkaConsumer::pauseOrResumePartitions(const TopicPartitions& topicPartitions, PauseOrResumeOperation op)
845 {
846  auto rk_tpos = rd_kafka_topic_partition_list_unique_ptr(createRkTopicPartitionList(topicPartitions));
847 
848  Error error{ (op == PauseOrResumeOperation::Pause) ?
849  rd_kafka_pause_partitions(getClientHandle(), rk_tpos.get()) : rd_kafka_resume_partitions(getClientHandle(), rk_tpos.get()) };
850  KAFKA_THROW_IF_WITH_ERROR(error);
851 
852  const char* opString = (op == PauseOrResumeOperation::Pause) ? "pause" : "resume";
853  int cnt = 0;
854  for (int i = 0; i < rk_tpos->cnt; ++i)
855  {
856  const rd_kafka_topic_partition_t& rk_tp = rk_tpos->elems[i];
857  if (rk_tp.err != RD_KAFKA_RESP_ERR_NO_ERROR)
858  {
859  KAFKA_API_DO_LOG(Log::Level::Err, "%s topic-partition[%s-%d] error[%s]", opString, rk_tp.topic, rk_tp.partition, rd_kafka_err2str(rk_tp.err));
860  }
861  else
862  {
863  KAFKA_API_DO_LOG(Log::Level::Notice, "%sd topic-partition[%s-%d]", opString, rk_tp.topic, rk_tp.partition, rd_kafka_err2str(rk_tp.err));
864  ++cnt;
865  }
866  }
867 
868  if (cnt == 0 && op == PauseOrResumeOperation::Pause)
869  {
870  std::string errMsg = std::string("No partition could be ") + opString + std::string("d among TopicPartitions[") + toString(topicPartitions) + std::string("]");
871  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG, errMsg));
872  }
873 }

899 // Rebalance Callback (for class instance)
900 inline void
901 KafkaConsumer::onRebalance(rd_kafka_resp_err_t err, rd_kafka_topic_partition_list_t* rk_partitions)
902 {
903  TopicPartitions tps = getTopicPartitions(rk_partitions);
904  std::string tpsStr = toString(tps);
905 
906  if (err != RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS && err != RD_KAFKA_RESP_ERR__REVOKE_PARTITIONS)
907  {
908  KAFKA_API_DO_LOG(Log::Level::Err, "unknown re-balance event[%d], topic-partitions[%s]", err, tpsStr.c_str());
909  return;
910  }
911 
912  // Initialize attribute for cooperative protocol
913  if (!_cooperativeEnabled)
914  {
915  if (const char* protocol = rd_kafka_rebalance_protocol(getClientHandle()))
916  {
917  _cooperativeEnabled = (std::string(protocol) == "COOPERATIVE");
918  }
919  }
920 
921  KAFKA_API_DO_LOG(Log::Level::Notice, "re-balance event triggered[%s], cooperative[%s], topic-partitions[%s]",
922  err == RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS ? "ASSIGN_PARTITIONS" : "REVOKE_PARTITIONS",
923  isCooperativeEnabled() ? "enabled" : "disabled",
924  tpsStr.c_str());
925 
926  // Remove the mark for pending event
927  if (_pendingEvent
928  && ((err == RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS && *_pendingEvent == PendingEvent::PartitionsAssignment)
929  || (err == RD_KAFKA_RESP_ERR__REVOKE_PARTITIONS && *_pendingEvent == PendingEvent::PartitionsRevocation)))
930  {
931  _pendingEvent.reset();
932  }
933 
934  PartitionsRebalanceEvent event = (err == RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS ?
935  (isCooperativeEnabled() ? PartitionsRebalanceEvent::IncrementalAssign : PartitionsRebalanceEvent::Assign)
936  : (isCooperativeEnabled() ? PartitionsRebalanceEvent::IncrementalUnassign : PartitionsRebalanceEvent::Revoke));
937 
938  if (err == RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS)
939  {
940  changeAssignment(event, tps);
941  }
942 
943  if (_rebalanceCb)
944  {
945  _rebalanceCb(err == RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS ? consumer::RebalanceEventType::PartitionsAssigned : consumer::RebalanceEventType::PartitionsRevoked,
946  tps);
947  }
948 
949  if (err == RD_KAFKA_RESP_ERR__REVOKE_PARTITIONS)
950  {
951  changeAssignment(event, isCooperativeEnabled() ? tps : TopicPartitions{});
952  }
953 }
