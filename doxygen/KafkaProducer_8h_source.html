270 inline Properties
271 KafkaProducer::validateAndReformProperties(const Properties& properties)
272 {
273  // Let the base class validate first
274  auto newProperties = KafkaClient::validateAndReformProperties(properties);
275 
276  // Check whether it's an available partitioner
277  const std::set<std::string> availPartitioners = {"murmur2_random", "murmur2", "random", "consistent", "consistent_random", "fnv1a", "fnv1a_random"};
278  auto partitioner = newProperties.getProperty(producer::Config::PARTITIONER);
279  if (partitioner && !availPartitioners.count(*partitioner))
280  {
281  std::string errMsg = "Invalid partitioner [" + *partitioner + "]! Valid options: ";
282  bool isTheFirst = true;
283  for (const auto& availPartitioner: availPartitioners)
284  {
285  errMsg += (std::string(isTheFirst ? (isTheFirst = false, "") : ", ") + availPartitioner);
286  }
287  errMsg += ".";
288 
289  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG, errMsg));
290  }
291 
292  // For "idempotence" feature
293  constexpr int KAFKA_IDEMP_MAX_INFLIGHT = 5;
294  const auto enableIdempotence = newProperties.getProperty(producer::Config::ENABLE_IDEMPOTENCE);
295  if (enableIdempotence && *enableIdempotence == "true")
296  {
297  if (const auto maxInFlight = newProperties.getProperty(producer::Config::MAX_IN_FLIGHT))
298  {
299  if (std::stoi(*maxInFlight) > KAFKA_IDEMP_MAX_INFLIGHT)
300  {
301  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG,\
302  "`max.in.flight` must be set <= " + std::to_string(KAFKA_IDEMP_MAX_INFLIGHT) + " when `enable.idempotence` is `true`"));
303  }
304  }
305 
306  if (const auto acks = newProperties.getProperty(producer::Config::ACKS))
307  {
308  if (*acks != "all" && *acks != "-1")
309  {
310  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG,\
311  "`acks` must be set to `all`/`-1` when `enable.idempotence` is `true`"));
312  }
313  }
314  }
315 
316  return newProperties;

330 inline void
331 KafkaProducer::send(const producer::ProducerRecord& record,
332  const producer::Callback& deliveryCb,
333  SendOption option,
334  ActionWhileQueueIsFull action)
335 {
336  auto deliveryCbOpaque = std::make_unique<DeliveryCbOpaque>(record.id(), deliveryCb);
337  auto queueFullAction = (isWithAutoEventsPolling() ? action : ActionWhileQueueIsFull::NoBlock);
338 
339  const auto* topic = record.topic().c_str();
340  const auto partition = record.partition();
341  const auto msgFlags = (static_cast<unsigned int>(option == SendOption::ToCopyRecordValue ? RD_KAFKA_MSG_F_COPY : 0)
342  | static_cast<unsigned int>(queueFullAction == ActionWhileQueueIsFull::Block ? RD_KAFKA_MSG_F_BLOCK : 0));
343  const auto* keyPtr = record.key().data();
344  const auto keyLen = record.key().size();
345  const auto* valuePtr = record.value().data();
346  const auto valueLen = record.value().size();
347 
348  auto* rk = getClientHandle();
349  auto* opaquePtr = deliveryCbOpaque.get();
350 
351  constexpr std::size_t VU_LIST_SIZE_WITH_NO_HEADERS = 6;
352  std::vector<rd_kafka_vu_t> rkVUs(VU_LIST_SIZE_WITH_NO_HEADERS + record.headers().size());
353 
354  std::size_t uvCount = 0;
355 
356  { // Topic
357  auto& vu = rkVUs[uvCount++];
358  vu.vtype = RD_KAFKA_VTYPE_TOPIC;
359  vu.u.cstr = topic;
360  }
361 
362  { // Partition
363  auto& vu = rkVUs[uvCount++];
364  vu.vtype = RD_KAFKA_VTYPE_PARTITION;
365  vu.u.i32 = partition;
366  }
367 
368  { // Message flags
369  auto& vu = rkVUs[uvCount++];
370  vu.vtype = RD_KAFKA_VTYPE_MSGFLAGS;
371  vu.u.i = static_cast<int>(msgFlags);
372  }
373 
374  { // Key
375  auto& vu = rkVUs[uvCount++];
376  vu.vtype = RD_KAFKA_VTYPE_KEY;
377  vu.u.mem.ptr = const_cast<void*>(keyPtr); // NOLINT
378  vu.u.mem.size = keyLen;
379  }
380 
381  { // Value
382  auto& vu = rkVUs[uvCount++];
383  vu.vtype = RD_KAFKA_VTYPE_VALUE;
384  vu.u.mem.ptr = const_cast<void*>(valuePtr); // NOLINT
385  vu.u.mem.size = valueLen;
386  }
387 
388  { // Opaque
389  auto& vu = rkVUs[uvCount++];
390  vu.vtype = RD_KAFKA_VTYPE_OPAQUE;
391  vu.u.ptr = opaquePtr;
392  }
393 
394  // Headers
395  for (const auto& header: record.headers())
396  {
397  auto& vu = rkVUs[uvCount++];
398  vu.vtype = RD_KAFKA_VTYPE_HEADER;
399  vu.u.header.name = header.key.c_str();
400  vu.u.header.val = header.value.data();
401  vu.u.header.size = header.value.size();
402  }
403 
404  assert(uvCount == rkVUs.size());
405 
406  Error sendResult{ rd_kafka_produceva(rk, rkVUs.data(), rkVUs.size()) };
407  KAFKA_THROW_IF_WITH_ERROR(sendResult);
408 
409  // KafkaProducer::deliveryCallback would delete the "opaque"
410  deliveryCbOpaque.release();
411 }

413 inline producer::RecordMetadata
414 KafkaProducer::syncSend(const producer::ProducerRecord& record)
415 {
416  Optional<Error> deliveryResult;
417  producer::RecordMetadata recordMetadata;
418  std::mutex mtx;
419  std::condition_variable delivered;
420 
421  auto deliveryCb = [&deliveryResult, &recordMetadata, &mtx, &delivered] (const producer::RecordMetadata& metadata, const Error& error) {
422  std::lock_guard<std::mutex> guard(mtx);
423 
424  deliveryResult = error;
425  recordMetadata = metadata;
426 
427  delivered.notify_one();
428  };
429 
430  send(record, deliveryCb);
431 
432  std::unique_lock<std::mutex> lock(mtx);
433  delivered.wait(lock, [&deliveryResult]{ return static_cast<bool>(deliveryResult); });
434 
435  KAFKA_THROW_IF_WITH_ERROR(*deliveryResult);
436 
437  return recordMetadata;
438 }
