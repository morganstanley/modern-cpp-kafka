332 KafkaClient::KafkaClient(ClientType clientType,
333  const Properties& properties,
334  const ConfigCallbacksRegister& extraConfigRegister,
335  EventsPollingOption eventsPollingOption)
336  : _eventsPollingOption(eventsPollingOption)
337 {
338  static const std::set<std::string> PRIVATE_PROPERTY_KEYS = { "max.poll.records" };
339 
340  // Save clientID
341  if (auto clientId = properties.getProperty(CLIENT_ID))
342  {
343  _clientId = *clientId;
344  _clientName = getClientTypeString(clientType) + "[" + _clientId + "]";
345  }
346 
347  // Init global logger
348  std::call_once(Global<>::initOnce, [](){ Global<>::logger = DefaultLogger; });
349 
350  // Save LogLevel
351  if (auto logLevel = properties.getProperty(LOG_LEVEL))
352  {
353  try
354  {
355  _logLevel = std::stoi(*logLevel);
356  }
357  catch (const std::exception& e)
358  {
359  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG, std::string("Invalid log_level[").append(*logLevel).append("], which must be an number!").append(e.what())));
360  }
361 
362  if (_logLevel < Log::Level::Emerg || _logLevel > Log::Level::Debug)
363  {
364  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG, std::string("Invalid log_level[").append(*logLevel).append("], which must be a value between 0 and 7!")));
365  }
366  }
367 
368  LogBuffer<LOG_BUFFER_SIZE> errInfo;
369 
370  auto rk_conf = rd_kafka_conf_unique_ptr(rd_kafka_conf_new());
371 
372  for (const auto& prop: properties.map())
373  {
374  // Those private properties are only available for `C++ wrapper`, not for librdkafka
375  if (PRIVATE_PROPERTY_KEYS.count(prop.first))
376  {
377  _properties.put(prop.first, prop.second);
378  continue;
379  }
380 
381  rd_kafka_conf_res_t result = rd_kafka_conf_set(rk_conf.get(), prop.first.c_str(), prop.second.c_str(), errInfo.str(), errInfo.capacity());
382  if (result == RD_KAFKA_CONF_OK)
383  {
384  _properties.put(prop.first, prop.second);
385  }
386  else
387  {
388  KAFKA_API_DO_LOG(Log::Level::Err, "failed to be initialized with property[%s:%s], result[%d]", prop.first.c_str(), prop.second.c_str(), result);
389  }
390  }
391 
392  // Save KafkaClient's raw pointer to the "opaque" field, thus we could fetch it later (for kinds of callbacks)
393  rd_kafka_conf_set_opaque(rk_conf.get(), this);
394 
395  // Log Callback
396  rd_kafka_conf_set_log_cb(rk_conf.get(), KafkaClient::logCallback);
397 
398  // Statistics Callback
399  rd_kafka_conf_set_stats_cb(rk_conf.get(), KafkaClient::statsCallback);
400 
401  // Error Callback
402  rd_kafka_conf_set_error_cb(rk_conf.get(), KafkaClient::errorCallback);
403 
404  // Other Callbacks
405  if (extraConfigRegister) extraConfigRegister(rk_conf.get());
406 
407  // Set client handler
408  _rk.reset(rd_kafka_new((clientType == ClientType::KafkaConsumer ? RD_KAFKA_CONSUMER : RD_KAFKA_PRODUCER),
409  rk_conf.release(), // rk_conf's ownship would be transferred to rk, after the "rd_kafka_new()" call
410  errInfo.clear().str(),
411  errInfo.capacity()));
412  KAFKA_THROW_IF_WITH_ERROR(Error(rd_kafka_last_error()));
413 
414  // Add brokers
415  auto brokers = properties.getProperty(BOOTSTRAP_SERVERS);
416  if (rd_kafka_brokers_add(getClientHandle(), brokers->c_str()) == 0)
417  {
418  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG,\
419  "No broker could be added successfully, BOOTSTRAP_SERVERS=[" + *brokers + "]"));
420  }
421 
422  _opened = true;
423 }

452 inline Optional<std::string>
453 KafkaClient::getProperty(const std::string& name) const
454 {
455  // Find it in pre-saved properties
456  if (auto property = _properties.getProperty(name)) return *property;
457 
458  constexpr int DEFAULT_BUF_SIZE = 512;
459 
460  const rd_kafka_conf_t* conf = rd_kafka_conf(getClientHandle());
461 
462  std::vector<char> valueBuf(DEFAULT_BUF_SIZE);
463  std::size_t valueSize = valueBuf.size();
464 
465  // Try with a default buf size. If could not find the property, return immediately.
466  if (rd_kafka_conf_get(conf, name.c_str(), valueBuf.data(), &valueSize) != RD_KAFKA_CONF_OK) return Optional<std::string>{};
467 
468  // If the default buf size is not big enough, retry with a larger one
469  if (valueSize > valueBuf.size())
470  {
471  valueBuf.resize(valueSize);
472  [[maybe_unused]] rd_kafka_conf_res_t result = rd_kafka_conf_get(conf, name.c_str(), valueBuf.data(), &valueSize);
473  assert(result == RD_KAFKA_CONF_OK);
474  }
475 
476  return std::string(valueBuf.data());
477 }

538 inline Optional<BrokerMetadata>
539 KafkaClient::fetchBrokerMetadata(const std::string& topic, std::chrono::milliseconds timeout, bool disableErrorLogging)
540 {
541  const rd_kafka_metadata_t* rk_metadata = nullptr;
542  // Here the input parameter for `all_topics` is `true`, since we want the `cgrp_update`
543  rd_kafka_resp_err_t err = rd_kafka_metadata(getClientHandle(), true, nullptr, &rk_metadata, convertMsDurationToInt(timeout));
544 
545  auto guard = rd_kafka_metadata_unique_ptr(rk_metadata);
546 
547  if (err != RD_KAFKA_RESP_ERR_NO_ERROR)
548  {
549  if (!disableErrorLogging)
550  {
551  KAFKA_API_DO_LOG(Log::Level::Err, "failed to get BrokerMetadata! error[%s]", rd_kafka_err2str(err));
552  }
553  return Optional<BrokerMetadata>{};
554  }
555 
556  const rd_kafka_metadata_topic* metadata_topic = nullptr;
557  for (int i = 0; i < rk_metadata->topic_cnt; ++i)
558  {
559  if (rk_metadata->topics[i].topic == topic)
560  {
561  metadata_topic = &rk_metadata->topics[i];
562  break;
563  }
564  }
565 
566  if (!metadata_topic || metadata_topic->err)
567  {
568  if (!disableErrorLogging)
569  {
570  if (!metadata_topic)
571  {
572  KAFKA_API_DO_LOG(Log::Level::Err, "failed to find BrokerMetadata for topic[%s]", topic.c_str());
573  }
574  else
575  {
576  KAFKA_API_DO_LOG(Log::Level::Err, "failed to get BrokerMetadata for topic[%s]! error[%s]", topic.c_str(), rd_kafka_err2str(metadata_topic->err));
577  }
578  }
579  return Optional<BrokerMetadata>{};
580  }
581 
582  // Construct the BrokerMetadata
583  BrokerMetadata metadata(metadata_topic->topic);
584  metadata.setOrigNodeName(rk_metadata->orig_broker_name ? std::string(rk_metadata->orig_broker_name) : "");
585 
586  for (int i = 0; i < rk_metadata->broker_cnt; ++i)
587  {
588  metadata.addNode(rk_metadata->brokers[i].id, rk_metadata->brokers[i].host, rk_metadata->brokers[i].port);
589  }
590 
591  for (int i = 0; i < metadata_topic->partition_cnt; ++i)
592  {
593  const rd_kafka_metadata_partition& metadata_partition = metadata_topic->partitions[i];
594 
595  Partition partition = metadata_partition.id;
596 
597  if (metadata_partition.err != 0)
598  {
599  if (!disableErrorLogging)
600  {
601  KAFKA_API_DO_LOG(Log::Level::Err, "got error[%s] while constructing BrokerMetadata for topic[%s]-partition[%d]", rd_kafka_err2str(metadata_partition.err), topic.c_str(), partition);
602  }
603 
604  continue;
605  }
606 
607  BrokerMetadata::PartitionInfo partitionInfo(metadata_partition.leader);
608 
609  for (int j = 0; j < metadata_partition.replica_cnt; ++j)
610  {
611  partitionInfo.addReplica(metadata_partition.replicas[j]);
612  }
613 
614  for (int j = 0; j < metadata_partition.isr_cnt; ++j)
615  {
616  partitionInfo.addInSyncReplica(metadata_partition.isrs[j]);
617  }
618 
619  metadata.addPartitionInfo(partition, partitionInfo);
620  }
621 
622  return metadata;
623 }
